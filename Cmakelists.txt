cmake_minimum_required(VERSION 3.10)
project(imguiTmpl)

set(CMAKE_CXX_STANDARD 11)

include_directories(./lib/glfw-3.3.8/include
                    ./lib/imgui
                    ./lib/imgui/backend
        )
link_directories(
        ./lib/glfw-3.3.8/lib
)

file(GLOB_RECURSE IMGUI_SRCS ./lib/imgui/*.cpp)
message(${IMGUI_SRCS})

#添加WIN32参数去除控制台界面了
add_executable(imguiTmpl WIN32 src/main.cpp ${IMGUI_SRCS}) 
target_link_libraries(imguiTmpl glfw3.a opengl32)
